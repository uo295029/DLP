      metadata⟦program⟧
      <call main>
      <halt>
      execute⟦definitioni⟧
execute⟦definition⟧ execute⟦varDefinition:definition → name:string type⟧ =
      error

execute⟦structDefinition:definition → name:string attributes:attribute*⟧ =
      error

execute⟦functionDefinition:definition → name:string params:varDefinition* type v:varDefinition* s:statement*⟧ =
      line(functionDefinition.start)
      functionDefinition.name<:>
      <enter > functionDefinition.varsSize
      execute⟦statementi⟧
      if (!hasReturn(statement*))
           <ret 0, > functionDefinition.varsSize <, > functionDefinition.paramsSize
execute⟦statement⟧ execute⟦if:statement → condition:expression yes:statement* no:statement*⟧ =
      line(if.start)
      string falseLabel = nextLabel()
      string endLabel = nextLabel()
      value⟦condition⟧
      <jz > falseLabel
      execute⟦yesi⟧
      <jmp > endLabel
      falseLabel
      execute⟦noi⟧
      endLabel

execute⟦while:statement → condition:expression yes:statement*⟧ =
      line(while.start)
      string conditionLabel = nextLabel()
      string endLabel = nextLabel()
      conditionLabel
      value⟦condition⟧
      <jz > endLabel
      execute⟦yesi⟧
      <jmp > conditionLabel
      endLabel

execute⟦read:statement → expression⟧ =
      line(read.end)
      address⟦expression⟧
      <in> suffix(expression.type)
      <store> suffix(expression.type)

execute⟦print:statement → expression*⟧ =
      line(print.end)
      if(!expression*.isEmpty(())) {
           value⟦expressioni⟧
           <out> suffix(expressioni.type)
      }

execute⟦printsp:statement → expression*⟧ =
      line(printsp.end)
      if(expression*.isEmpty(())) {
           <pushb 32>
           <outb>
      }
      else {
           value⟦expressioni⟧
           <out> suffix(expressioni.type)
           <pushb 32>
           <outb>
      }

execute⟦println:statement → expression*⟧ =
      line(println.end)

           <pushb 10>
           <outb>
      }
      else {
           value⟦expressioni⟧
           <out> suffix(expressioni.type)

           <outb>
      }

execute⟦assignment:statement → left:expression right:expression⟧ =
      line(assignment.end)
      address⟦left⟧
      value⟦right⟧
      <store> suffix(left.type)

execute⟦functionCallS:statement → name:string params:expression*⟧ =
      line(functionCallS.end)
      value⟦paramsi⟧
      <call > name
      if (!functionCallS.functionDefinition.type instanceof VoidType)
           <pop> suffix(functionCallS.functionDefinition.type)

execute⟦return:statement → expression?⟧ =
      line(return.end)
      if (expression.isPresent()) {
           value⟦expression⟧
           <ret > expression.type.bytes <, > return.functionReturn.varsSize <, > return.functionReturn.parametersSize
      }
      else {
           <ret 0, > return.functionReturn.varsSize <, > return.functionReturn.paramsSize
      }

metadata⟦definition⟧ metadata⟦varDefinition:definition → name:string type⟧ =
      switch (varDefinition.scope) {
      case 0:
           #GLOBAL name:maplType(type)
           break;
      case 1:
           #PARAM name:maplType(type)
           break;
      case 2:
           #LOCAL name:maplType(type)
           break;
      }

metadata⟦structDefinition:definition → name:string attributes:attribute*⟧ =
      #TYPE name: {
           metadata⟦attributei⟧
      }

metadata⟦functionDefinition:definition → name:string params:varDefinition* type v:varDefinition* s:statement*⟧ =
      #FUNC name
      metadata⟦paramsi⟧
      #RET maplType(type)
      metadata⟦vi⟧
